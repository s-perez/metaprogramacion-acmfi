Ejemplo de decoradores

  • Añadir Caché a una operación

>> def cache(f):
	almacen = {}
	def funcion(*args, **kwargs):
		key = hash(args)
		key2 = hash(tuple(kwargs.iteritems()))
		try:
			resultado = almacen[key, key2]
		except KeyError:
			almacen[key,key2] = f(*args,**kwargs)
			resultado = almacen[key, key2]
		return resultado
	return funcion

>> @cache
def suma(a,b):
	return a + b
