>> def cache(f):
	almacen = {}
	def funcion(*args, **kwargs):
		key = hash(args)
		key2 = hash(tuple(kwargs.iteritems()))
		try:
			resultado = almacen[key, key2]
		except KeyError:
			almacen[key,key2] = f(*args,**kwargs)
			resultado = almacen[key, key2]
		return resultado
	return funcion

¿Y cómo busca Python?

cuando llamas a un método, estás llamando a objeto.__getattribute__("nombre del metodo")

Vamos a sobreescribirlo!!


class ShopQueries():
	"""A web for my own shop"""
	def build_get(var):
		def funcion(self):
			return getattr(self,var,None)
		return funcion
	def __getattribute__(self, name):
		if not hasattr(self, name):
			if name[:3] == 'get':
				setattr(ShopQueries,"get%s" % name,build_get(name[3:]))
			return super(ShopQueries,self).__getattribute__(name)


>> s = ShopQueries()
>> s.getItems()
